cm-chessboard

Example: Context input, mark squares

Toggle marker on context input. First board, context input only. Second board, move input and context input enabled.


const board1 = new Chessboard(document.getElementById("board1"), {
    position: "start",
    events: {
        contextInput: (square) => {
            const markersOnSquare = board1.getMarkers(square, MARKER_TYPE.emphasize);
            if(markersOnSquare.length > 0) {
                board1.removeMarkers(square, MARKER_TYPE.emphasize);
            } else {
                board1.addMarker(square);
            }
        }
    }
});
const board2 = new Chessboard(document.getElementById("board2"), {
    position: "rn2k1r1/ppp1pp1p/3p2p1/5bn1/P7/2N2B2/1PPPPP2/2BNK1RR",
    orientation: COLOR.black,
    moveInputMode: MOVE_INPUT_MODE.dragPiece,
    events: {
        contextInput: function(square) {
            const markersOnSquare = board2.getMarkers(square, MARKER_TYPE.emphasize);
            if(markersOnSquare.length > 0) {
                board2.removeMarkers(square, MARKER_TYPE.emphasize);
            } else {
                board2.addMarker(square);
            }
        }
    }
}, (board) => {
    board.enableMoveInput(COLOR.white, true);
    board.enableMoveInput(COLOR.black, true);
});
