cm-chessboard

Example: Input enabled without validation

Input enabled on both sides, without move validation.

const board = new Chessboard(document.getElementById("board"), {
    position: "start",
    moveInputMode: MOVE_INPUT_MODE.dragFigure,
    events: {
        moveInputStart: (square) => {
            log(`moveInputStart: ${square}`);
            return true;
        },
        moveInputDone: (squareFrom, squareTo) => {
            log(`moveInputDone: ${squareFrom}-${squareTo}`);
            return true;
        }
    }
}, () => {
    board.enableInput(COLOR.white, true);
    board.enableInput(COLOR.black, true);
});
